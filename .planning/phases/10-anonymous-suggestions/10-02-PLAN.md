    ```typescript
    export const anonymousSuggestionsApi = {
      createSession: (projectId: string) =>
        api.post<AnonymousSessionCreateResponse>(
          `/api/v1/projects/${projectId}/suggestions/anonymous/sessions`,
        ),

      save: (projectId: string, sessionId: string, data: SuggestionSavePayload, anonymousToken: string) =>
        api.put<SuggestionSaveResponse>(
          `/api/v1/projects/${projectId}/suggestions/anonymous/sessions/${sessionId}/save`,
          data,
          { headers: { "X-Anonymous-Token": anonymousToken } },
        ),

      submit: (projectId: string, sessionId: string, data: AnonymousSubmitPayload, anonymousToken: string) =>
        api.post<SuggestionSubmitResponse>(
          `/api/v1/projects/${projectId}/suggestions/anonymous/sessions/${sessionId}/submit`,
          data,
          { headers: { "X-Anonymous-Token": anonymousToken } },
        ),

      discard: (projectId: string, sessionId: string, anonymousToken: string) =>
        api.post<void>(
          `/api/v1/projects/${projectId}/suggestions/anonymous/sessions/${sessionId}/discard`,
          undefined,
          { headers: { "X-Anonymous-Token": anonymousToken } },
        ),

      beacon: (projectId: string, sessionId: string, content: string, anonymousToken: string) => {
        const url = `${API_BASE}/api/v1/projects/${projectId}/suggestions/anonymous/beacon?token=${encodeURIComponent(anonymousToken)}`;
        const payload = JSON.stringify({ session_id: sessionId, content });
        const blob = new Blob([payload], { type: "application/json" });
        return navigator.sendBeacon(url, blob);
      },
    };
    ```

    **1. Create `lib/hooks/useAnonymousSuggestion.ts`:**

    This hook manages the full anonymous suggestion session lifecycle, mirroring `useSuggestionSession` but using anonymous tokens instead of Bearer tokens.

    ```typescript
    export interface UseAnonymousSuggestionReturn {
      sessionId: string | null;
      branch: string | null;
      anonymousToken: string | null;
      changesCount: number;
      status: SuggestionStatus; // reuse from useSuggestionSession
      error: string | null;
      entitiesModified: string[];
      isActive: boolean;
      startSession: () => Promise<void>;
      saveToSession: (content: string, entityIri: string, entityLabel: string) => Promise<void>;
      submitSession: (summary?: string, submitterName?: string, submitterEmail?: string) => Promise<void>;
      discardSession: () => Promise<void>;
    }

    interface UseAnonymousSuggestionOptions {
      projectId: string;
      onSubmitted?: (prNumber: number, prUrl: string | null) => void;
      onError?: (msg: string) => void;
    }
    ```

    Implementation details:
    - On mount, check `useAnonymousTokenStore.getToken(projectId)` for an existing active session. If found, restore `sessionId`, `branch`, `anonymousToken`, set status to "active".
    - `startSession()`: call `anonymousSuggestionsApi.createSession(projectId)`, store the returned token in `useAnonymousTokenStore.setToken(projectId, ...)`, set state.
    - `saveToSession()`: call `anonymousSuggestionsApi.save(...)` with the stored anonymous token.
    - `submitSession()`: call `anonymousSuggestionsApi.submit(...)` with summary + submitter info + `website: ""` (honeypot always empty). On success, call `useAnonymousTokenStore.clearToken(projectId)` and reset state.
    - `discardSession()`: call `anonymousSuggestionsApi.discard(...)`, clear token store, reset state.
    - Export `SuggestionStatus` type from `useSuggestionSession.ts` and reuse it (or re-export if already exported).
