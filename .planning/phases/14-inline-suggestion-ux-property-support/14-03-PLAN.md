**1. Editor page (`app/projects/[id]/editor/page.tsx`)** — direct entity creation for accepted child suggestions (per D-07):

The existing `handleEntityConfirm` shows how entity creation works: it calls `generateTurtleSnippet()` to produce Turtle text, inserts it into the source editor, and calls `addOptimisticNode()` to add the node to the tree instantly. The new `handleAddSuggestedChild` MUST follow this same pattern — calling the creation API directly WITHOUT opening AddEntityDialog.

```typescript
// Track accepted suggestion IRIs for sparkle badges (per D-07)
const [acceptedSuggestionIris, setAcceptedSuggestionIris] = useState<Set<string>>(new Set());

const handleAddSuggestedChild = useCallback((iri: string, label: string, parentIri: string) => {
  // Generate Turtle snippet for the new entity (same as handleEntityConfirm)
  const snippet = generateTurtleSnippet({
    iri,
    label,
    entityType: "class",
    parentIri,
    ontologyPrefix,
    ontologyNamespace,
  });

  // Insert into source
  if (sourceEditorRef.current) {
    sourceEditorRef.current.insertAtEnd(snippet);
    setSourceContent(sourceEditorRef.current.getValue());
  } else if (sourceContent) {
    setSourceContent((prev) => prev + snippet);
  }

  // Add to tree immediately (per D-07)
  addOptimisticNode(iri, label, parentIri);

  // Track as accepted-suggestion IRI for sparkle badge
  setAcceptedSuggestionIris((prev) => new Set(prev).add(iri));
}, [ontologyPrefix, ontologyNamespace, addOptimisticNode, sourceContent, setSourceContent]);
```
