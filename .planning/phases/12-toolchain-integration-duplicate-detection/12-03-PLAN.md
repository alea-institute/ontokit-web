   ```python
   async def semantic_search_all_branches(
       self,
       project_id: UUID,
       query: str,
       limit: int = 20,
       threshold: float = 0.3,
   ) -> list[SemanticSearchResultWithBranch]:
       """Search across ALL branches for a project — needed for cross-branch duplicate detection (D-08).
       
       Unlike semantic_search() which filters by branch, this queries across all branches
       to catch parallel work collisions between users on different suggestion branches.
       """
       if Vector is None:
           raise RuntimeError("pgvector is not installed. Semantic search requires pgvector.")
       
       # Get embedding provider config
       config = await self._get_config(project_id)
       if not config:
           return []
       
       provider = get_embedding_provider(config.provider, config.model_name, ...)
       query_vec = await provider.embed_text(query)
       
       # Query across ALL branches (no branch filter)
       query_str = text("""
           SELECT entity_iri, label, entity_type, branch, deprecated,
                  1 - (embedding <=> :query_vec::vector) AS score
           FROM entity_embeddings
           WHERE project_id = :pid
             AND (1 - (embedding <=> :query_vec::vector)) >= :threshold
           ORDER BY embedding <=> :query_vec::vector
           LIMIT :lim
       """)
       result = await self._db.execute(
           query_str,
           {"pid": str(project_id), "query_vec": str(query_vec), "threshold": threshold, "lim": limit},
       )
       rows = result.fetchall()
       return [
           SemanticSearchResultWithBranch(
               iri=row.entity_iri,
               label=row.label or "",
               entity_type=row.entity_type,
               score=round(float(row.score), 4),
               deprecated=row.deprecated,
               branch=row.branch,
           )
           for row in rows
       ]
   ```
   
   Follow the exact same pattern as the existing `semantic_search()` method for embedding provider initialization, vector embedding, and SQL execution. The ONLY difference is removing the `branch = :br` WHERE clause.

3. **Create `ontokit/services/startup_checks.py`**:
   ```python
   """Startup health checks — run once during app lifespan initialization."""
   
   import logging
   from datetime import UTC, datetime, timedelta
   
   from sqlalchemy import select
   from sqlalchemy.ext.asyncio import AsyncSession
   
   from ontokit.core.database import async_session
   from ontokit.models.embedding import EmbeddingJob, ProjectEmbeddingConfig
   
   logger = logging.getLogger(__name__)
   
   # If last full embed is older than this, trigger rebuild
   STALE_THRESHOLD = timedelta(hours=24)
   
   
   async def check_and_trigger_embedding_rebuilds() -> None:
       """Check all projects for stale embedding indexes and enqueue rebuilds.
       
       Called during app startup (D-05). A stale index is one where
       last_full_embed_at is None or older than STALE_THRESHOLD.
       """
       async with async_session() as db:
           configs = (await db.execute(
               select(ProjectEmbeddingConfig).where(
                   ProjectEmbeddingConfig.auto_embed_on_save.is_(True)
               )
           )).scalars().all()
           
           now = datetime.now(UTC)
           rebuild_count = 0
           
           for config in configs:
               is_stale = (
                   config.last_full_embed_at is None
                   or (now - config.last_full_embed_at) > STALE_THRESHOLD
               )
               if not is_stale:
                   continue
               
               # Check for active job
               active = (await db.execute(
                   select(EmbeddingJob).where(
                       EmbeddingJob.project_id == config.project_id,
                       EmbeddingJob.status.in_(["pending", "processing"]),
                   )
               )).scalars().first()
               
               if active:
                   logger.info("Project %s has active embedding job, skipping startup rebuild", config.project_id)
                   continue
               
               # Enqueue rebuild
               try:
                   import uuid
                   from ontokit.core.redis import get_arq_pool
                   
                   job_id = uuid.uuid4()
                   new_job = EmbeddingJob(id=job_id, project_id=config.project_id, branch="main", status="pending")
                   db.add(new_job)
                   await db.commit()
                   
                   pool = await get_arq_pool()
                   await pool.enqueue_job("run_embedding_generation_task", str(config.project_id), "main", str(job_id))
                   rebuild_count += 1
                   logger.info("Startup: enqueued embedding rebuild for project %s", config.project_id)
               except Exception:
                   logger.exception("Failed to enqueue startup rebuild for project %s", config.project_id)
           
           if rebuild_count:
               logger.info("Startup: triggered %d embedding rebuild(s)", rebuild_count)
           else:
               logger.info("Startup: all embedding indexes are fresh")
   ```
   
   Check how `async_session` is exported from `ontokit.core.database` — it may be named differently (e.g., `get_async_session`, `SessionLocal`, or use `async_sessionmaker`). Match the existing pattern.
