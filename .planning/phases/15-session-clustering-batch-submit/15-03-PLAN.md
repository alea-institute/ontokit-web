Create `lib/hooks/useShardDragDrop.ts`:

```typescript
"use client";

import { useCallback } from "react";
import { useSensor, useSensors, PointerSensor, KeyboardSensor } from "@dnd-kit/core";
import type { DragEndEvent } from "@dnd-kit/core";
import { useShardPreviewStore } from "@/lib/stores/shardPreviewStore";

interface ShardDragData {
  type: "entity" | "shard";
  entityIri?: string;
  fromShardId?: string;
  shardId?: string;
  fromPrId?: string;
}

export function useShardDragDrop() {
  const moveEntity = useShardPreviewStore((s) => s.moveEntity);
  const moveShard = useShardPreviewStore((s) => s.moveShard);

  const pointerSensor = useSensor(PointerSensor, {
    activationConstraint: { distance: 8 },
  });
  const keyboardSensor = useSensor(KeyboardSensor);
  const sensors = useSensors(pointerSensor, keyboardSensor);

  const handleDragEnd = useCallback(
    (event: DragEndEvent) => {
      const { active, over } = event;
      if (!over) return;

      const data = active.data.current as ShardDragData | undefined;
      if (!data) return;

      if (data.type === "entity" && data.entityIri && data.fromShardId) {
        const toShardId = String(over.id);
        if (toShardId === data.fromShardId) return; // no-op
        moveEntity(data.entityIri, data.fromShardId, toShardId);
      } else if (data.type === "shard" && data.shardId && data.fromPrId) {
        const toPrId = String(over.id);
        if (toPrId === data.fromPrId) return; // no-op
        moveShard(data.shardId, data.fromPrId, toPrId);
      }
    },
    [moveEntity, moveShard],
  );

  return { sensors, handleDragEnd };
}
```

**DndContext:**

- Use `sensors` and `handleDragEnd` from `useShardDragDrop()` hook
- The DndContext is the modal's own — NOT nested inside the tree's DndContext (modal is a fixed overlay, so no nesting issue per RESEARCH.md Pitfall 3)

**Derived values:**
- `allShardIds`: computed from `Object.keys(shards)`
- `shardLabelMap`: computed from `Object.fromEntries(Object.entries(shards).map(([id, s]) => [id, s.label]))`

**Phase B: Submitting**

When user clicks "Submit suggestions":
1. Set phase to `"submitting"`
2. Build `BatchSubmitRequest` from current store state:
   - Map `prGroupOrder` → `pr_groups[]`, each with `shards[]` from `prGroup.shardIds.map(id => ({ id, label: shards[id].label, entity_iris: shards[id].entityIris }))`
   - Attach `notes` if non-empty
3. Build progress steps: ["Submitting suggestions..."] + one step per PR group ("Creating PR {N}/{M}...")
4. Call `suggestionsApi.batchSubmit(projectId, sessionId, request, accessToken)`
5. On response: set phase to `"complete"`, store `BatchSubmitResponse`

Progress bar replaces the body content. Footer is hidden during submit.

**Phase C: Complete**

Render `ShardSubmitComplete` with results from `BatchSubmitResponse`.
- `onRetryFailed`: re-run `batchSubmit` with only the failed PR groups. Build a new `BatchSubmitRequest` containing only the `pr_groups` whose index matches a failed result.
- `onDone`: call `useShardPreviewStore().clear()`, call `onBatchSubmitted(results)`, call `onClose()`

**Animation contract (per UI-SPEC):**
- Modal open: `animate-in fade-in zoom-in-95 duration-150 ease-out` (Tailwind)
- Modal close: handled by unmount (parent controls rendering)
